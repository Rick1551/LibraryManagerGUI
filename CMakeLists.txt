cmake_minimum_required(VERSION 3.16)

project(LibraryManagerGUI VERSION 0.1 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(Qt6 REQUIRED COMPONENTS Quick)
find_package(Qt6 REQUIRED COMPONENTS Core)
find_package(Qt6 REQUIRED COMPONENTS Core)
find_package(Qt6 REQUIRED COMPONENTS Core)
find_package(Qt6 REQUIRED COMPONENTS Widgets)

qt_standard_project_setup(REQUIRES 6.8)

qt_add_executable(appLibraryManagerGUI
    main.cpp
)

qt_add_qml_module(appLibraryManagerGUI
    URI LibraryManagerGUI
    VERSION 1.0
    QML_FILES
        Main.qml
        SOURCES Book.h Book.cpp
        SOURCES Borrow.h Borrow.cpp
        SOURCES Users.h Users.cpp
        SOURCES LibraryManager.h LibraryManager.cpp
)

# Qt for iOS sets MACOSX_BUNDLE_GUI_IDENTIFIER automatically since Qt 6.1.
# If you are developing for iOS or macOS you should consider setting an
# explicit, fixed bundle identifier manually though.
set_target_properties(appLibraryManagerGUI PROPERTIES
#    MACOSX_BUNDLE_GUI_IDENTIFIER com.example.appLibraryManagerGUI
    MACOSX_BUNDLE_BUNDLE_VERSION ${PROJECT_VERSION}
    MACOSX_BUNDLE_SHORT_VERSION_STRING ${PROJECT_VERSION_MAJOR}.${PROJECT_VERSION_MINOR}
    MACOSX_BUNDLE TRUE
    WIN32_EXECUTABLE TRUE
)

target_link_libraries(appLibraryManagerGUI
    PRIVATE Qt6::Quick
)
target_link_libraries(appLibraryManagerGUI PRIVATE Qt6::Core)
target_link_libraries(appLibraryManagerGUI PRIVATE Qt6::Core)
target_link_libraries(appLibraryManagerGUI PRIVATE Qt6::Core)
target_link_libraries(appLibraryManagerGUI PRIVATE Qt6::Widgets)

include(GNUInstallDirs)
install(TARGETS appLibraryManagerGUI
    BUNDLE DESTINATION .
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
)
